#42 HARD

Trapping Rain Water

Array Two Pointers Dynamic Programming Stack

題目描述

給定 n 個非負整數表示每個寬度為 1 的柱子的高度圖，計算按此排列的柱子，下雨之後能接多少雨水。

範例

範例 1:

輸入：height = [0,1,0,2,1,0,1,3,2,1,2,1]
輸出：6
解釋：上面是由陣列 [0,1,0,2,1,0,1,3,2,1,2,1] 表示的高度圖，
在這種情況下，可以接 6 個單位的雨水（藍色部分表示雨水）。

視覺化：
   █
   █ █   █
 █ ██ █ ███ █ █
[0,1,0,2,1,0,1,3,2,1,2,1]

範例 2:

輸入：height = [4,2,0,3,2,5]
輸出：9

限制條件

  • n == height.length
  • 1 <= n <= 2 * 104
  • 0 <= height[i] <= 105

解題思路

核心觀察

對於每個位置 i，能接的雨水量 = min(左邊最高, 右邊最高) - height[i]

方法一：雙指標（推薦）

使用左右兩個指標，維護左右兩邊的最大高度：

  1. 從兩端向中間移動
  2. 較矮的一邊決定了能接多少水
  3. 更新最大高度並計算水量
  • 時間複雜度：O(n)
  • 空間複雜度：O(1)

方法二：動態規劃

預先計算每個位置左右兩邊的最大高度。

  • 時間複雜度：O(n)
  • 空間複雜度：O(n)

方法三：單調堆疊

使用堆疊計算每個凹槽能接的水。

  • 時間複雜度：O(n)
  • 空間複雜度：O(n)

程式碼實作

Python 解法

class Solution:
    def trap(self, height: List[int]) -> int:
        """
        雙指標解法 - 最優解
        
        時間複雜度：O(n)
        空間複雜度：O(1)
        """
        if not height:
            return 0
        
        left, right = 0, len(height) - 1
        left_max = right_max = 0
        water = 0
        
        while left < right:
            if height[left] < height[right]:
                # 左邊較矮，處理左邊
                if height[left] >= left_max:
                    left_max = height[left]
                else:
                    water += left_max - height[left]
                left += 1
            else:
                # 右邊較矮，處理右邊
                if height[right] >= right_max:
                    right_max = height[right]
                else:
                    water += right_max - height[right]
                right -= 1
        
        return water


# 方法二：動態規劃
class Solution2:
    def trap(self, height: List[int]) -> int:
        """
        動態規劃解法
        
        時間複雜度：O(n)
        空間複雜度：O(n)
        """
        if not height:
            return 0
        
        n = len(height)
        
        # 計算每個位置左邊的最大高度
        left_max = [0] * n
        left_max[0] = height[0]
        for i in range(1, n):
            left_max[i] = max(left_max[i-1], height[i])
        
        # 計算每個位置右邊的最大高度
        right_max = [0] * n
        right_max[n-1] = height[n-1]
        for i in range(n-2, -1, -1):
            right_max[i] = max(right_max[i+1], height[i])
        
        # 計算總水量
        water = 0
        for i in range(n):
            water += min(left_max[i], right_max[i]) - height[i]
        
        return water


# 方法三：單調堆疊
class Solution3:
    def trap(self, height: List[int]) -> int:
        """
        單調堆疊解法
        
        時間複雜度：O(n)
        空間複雜度：O(n)
        """
        water = 0
        stack = []  # 存儲索引
        
        for i in range(len(height)):
            # 當前高度大於堆疊頂部，可以形成凹槽
            while stack and height[i] > height[stack[-1]]:
                top = stack.pop()
                
                if not stack:
                    break
                
                # 計算凹槽的寬度和高度
                distance = i - stack[-1] - 1
                bounded_height = min(height[i], height[stack[-1]]) - height[top]
                water += distance * bounded_height
            
            stack.append(i)
        
        return water

C++ 解法

#include <vector>
#include <algorithm>
using namespace std;

class Solution {
public:
    int trap(vector<int>& height) {
        if (height.empty()) return 0;
        
        int left = 0, right = height.size() - 1;
        int leftMax = 0, rightMax = 0;
        int water = 0;
        
        while (left < right) {
            if (height[left] < height[right]) {
                if (height[left] >= leftMax) {
                    leftMax = height[left];
                } else {
                    water += leftMax - height[left];
                }
                left++;
            } else {
                if (height[right] >= rightMax) {
                    rightMax = height[right];
                } else {
                    water += rightMax - height[right];
                }
                right--;
            }
        }
        
        return water;
    }
};

        int water = 0;

                    leftMax = height[left];
                } else {
                    water += leftMax - height[left];
                }
                left++;
            } else {
                if (height[right] >= rightMax) {

            }
        }
        
        return water;
    }
}
