#4 HARD

Median of Two Sorted Arrays

Array Binary Search Divide and Conquer

題目描述

給定兩個大小分別為 m 和 n 的正序（從小到大）陣列 nums1 和 nums2。請你找出並返回這兩個正序陣列的中位數。

算法的時間複雜度應該為 O(log (m+n))。

範例

範例 1:

輸入：nums1 = [1,3], nums2 = [2]
輸出：2.00000
解釋：合併陣列 = [1,2,3]，中位數 2

範例 2:

輸入：nums1 = [1,2], nums2 = [3,4]
輸出：2.50000
解釋：合併陣列 = [1,2,3,4]，中位數 (2 + 3) / 2 = 2.5

限制條件

  • nums1.length == m
  • nums2.length == n
  • 0 <= m <= 1000
  • 0 <= n <= 1000
  • 1 <= m + n <= 2000
  • -106 <= nums1[i], nums2[i] <= 106

解題思路

核心思想

中位數將陣列分為兩部分，左邊所有元素 ≤ 右邊所有元素。

方法一：二分搜尋（推薦）

在較短的陣列上進行二分搜尋：

  1. 確保 nums1 是較短的陣列
  2. 在 nums1 中二分搜尋分割位置
  3. 根據 nums1 的分割位置計算 nums2 的分割位置
  4. 檢查分割是否有效：左邊最大 ≤ 右邊最小
  5. 根據總長度奇偶性計算中位數
  • 時間複雜度：O(log(min(m, n)))
  • 空間複雜度：O(1)

方法二：合併後排序（不符合要求）

合併兩個陣列並排序，找中位數。

  • 時間複雜度：O(m + n)
  • 不符合 O(log(m+n)) 的要求

程式碼實作

Python 解法

class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        """
        二分搜尋解法
        
        時間複雜度：O(log(min(m, n)))
        空間複雜度：O(1)
        """
        # 確保 nums1 是較短的陣列
        if len(nums1) > len(nums2):
            nums1, nums2 = nums2, nums1
        
        m, n = len(nums1), len(nums2)
        left, right = 0, m
        
        while left <= right:
            # nums1 的分割位置
            partition1 = (left + right) // 2
            # nums2 的分割位置
            partition2 = (m + n + 1) // 2 - partition1
            
            # 左邊最大值
            max_left1 = float('-inf') if partition1 == 0 else nums1[partition1 - 1]
            max_left2 = float('-inf') if partition2 == 0 else nums2[partition2 - 1]
            
            # 右邊最小值
            min_right1 = float('inf') if partition1 == m else nums1[partition1]
            min_right2 = float('inf') if partition2 == n else nums2[partition2]
            
            # 找到正確的分割
            if max_left1 <= min_right2 and max_left2 <= min_right1:
                # 總長度為奇數
                if (m + n) % 2 == 1:
                    return max(max_left1, max_left2)
                # 總長度為偶數
                else:
                    return (max(max_left1, max_left2) + min(min_right1, min_right2)) / 2
            
            # 調整分割位置
            elif max_left1 > min_right2:
                right = partition1 - 1
            else:
                left = partition1 + 1
        
        raise ValueError("Input arrays are not sorted")


# 方法二：合併後找中位數（O(m+n)，不符合要求）
class Solution2:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        """
        合併陣列解法
        
        時間複雜度：O(m + n)
        空間複雜度：O(m + n)
        """
        merged = []
        i, j = 0, 0
        
        # 合併兩個有序陣列
        while i < len(nums1) and j < len(nums2):
            if nums1[i] < nums2[j]:
                merged.append(nums1[i])
                i += 1
            else:
                merged.append(nums2[j])
                j += 1
        
        merged.extend(nums1[i:])
        merged.extend(nums2[j:])
        
        # 計算中位數
        n = len(merged)
        if n % 2 == 1:
            return merged[n // 2]
        else:
            return (merged[n // 2 - 1] + merged[n // 2]) / 2

C++ 解法

#include <vector>
#include <algorithm>
#include <climits>
using namespace std;

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        if (nums1.size() > nums2.size()) {
            return findMedianSortedArrays(nums2, nums1);
        }
        
        int m = nums1.size();
        int n = nums2.size();
        int left = 0, right = m;
        
        while (left <= right) {
            int partition1 = (left + right) / 2;
            int partition2 = (m + n + 1) / 2 - partition1;
            
            int maxLeft1 = (partition1 == 0) ? INT_MIN : nums1[partition1 - 1];
            int maxLeft2 = (partition2 == 0) ? INT_MIN : nums2[partition2 - 1];
            
            int minRight1 = (partition1 == m) ? INT_MAX : nums1[partition1];
            int minRight2 = (partition2 == n) ? INT_MAX : nums2[partition2];
            
            if (maxLeft1 <= minRight2 && maxLeft2 <= minRight1) {
                if ((m + n) % 2 == 1) {
                    return max(maxLeft1, maxLeft2);
                } else {
                    return (max(maxLeft1, maxLeft2) + min(minRight1, minRight2)) / 2.0;
                }
            } else if (maxLeft1 > minRight2) {
                right = partition1 - 1;
            } else {
                left = partition1 + 1;
            }
        }
        
        return 0.0;
    }
};

class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        if (nums1.length > nums2.length) {
            return findMedianSortedArrays(nums2, nums1);
        }
        
        int m = nums1.length;
        int n = nums2.length;
        int left = 0, right = m;
        
        while (left <= right) {
            int partition1 = (left + right) / 2;
            int partition2 = (m + n + 1) / 2 - partition1;
            
            int maxLeft1 = (partition1 == 0) ? Integer.MIN_VALUE : nums1[partition1 - 1];
            int maxLeft2 = (partition2 == 0) ? Integer.MIN_VALUE : nums2[partition2 - 1];
            
            int minRight1 = (partition1 == m) ? Integer.MAX_VALUE : nums1[partition1];
            int minRight2 = (partition2 == n) ? Integer.MAX_VALUE : nums2[partition2];
            
            if (maxLeft1 <= minRight2 && maxLeft2 <= minRight1) {
                if ((m + n) % 2 == 1) {
                    return Math.max(maxLeft1, maxLeft2);
                } else {
                    return (Math.max(maxLeft1, maxLeft2) + Math.min(minRight1, minRight2)) / 2.0;
                }
            } else if (maxLeft1 > minRight2) {
                right = partition1 - 1;
            } else {
                left = partition1 + 1;
            }
        }
        
        return 0.0;
    }
}

算法視覺化

以 nums1 = [1,3], nums2 = [2,4,5,6] 為例：

合併後：[1, 2, 3, 4, 5, 6]
中位數：(3 + 4) / 2 = 3.5

分割示意：
nums1: [1, 3] | []
nums2: [2] | [4, 5, 6]

左半部分：[1, 2, 3]
右半部分：[4, 5, 6]

檢查：max(左) ≤ min(右)
      3 ≤ 4 ✓

中位數 = (max(3, 2) + min(INF, 4)) / 2 = 3.5

為什麼是 O(log(min(m,n)))

我們在較短的陣列上進行二分搜尋：

  • 每次迭代將搜尋空間減半
  • 搜尋空間大小為 min(m, n)
  • 因此時間複雜度為 O(log(min(m, n)))
  • 這優於 O(log(m+n)) 的要求
