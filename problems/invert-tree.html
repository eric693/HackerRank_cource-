#226 EASY

Invert Binary Tree

Tree DFS BFS

題目描述

給定一個二元樹的根節點 root，翻轉這棵二元樹，並返回其根節點。

範例

範例 1:

輸入：root = [4,2,7,1,3,6,9]
輸出：[4,7,2,9,6,3,1]

原樹:           翻轉後:
     4              4
   /   \          /   \
  2     7        7     2
 / \   / \      / \   / \
1   3 6   9    9   6 3   1

範例 2:

輸入：root = [2,1,3]
輸出：[2,3,1]

原樹:      翻轉後:
  2          2
 / \        / \
1   3      3   1

範例 3:

輸入：root = []
輸出：[]

限制條件

  • 樹中節點數目範圍在 [0, 100] 內
  • -100 <= Node.val <= 100

解題思路

遞迴法（DFS）- 推薦

最簡潔的解法，利用遞迴交換左右子樹：

  1. 如果節點為空，直接返回
  2. 交換當前節點的左右子節點
  3. 遞迴地翻轉左子樹和右子樹
  • 時間複雜度：O(n) - 訪問每個節點一次
  • 空間複雜度：O(h) - h 為樹的高度，遞迴堆疊

迭代法（BFS）

使用佇列進行層序遍歷，逐個交換節點的左右子節點。

  • 時間複雜度：O(n)
  • 空間複雜度：O(n) - 佇列空間

程式碼實作

Python 解法

from collections import deque
from typing import Optional

class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

class Solution:
    def invertTree(self, root: Optional[TreeNode]) -> Optional[TreeNode]:
        """
        遞迴法 - 最簡潔的解法
        
        時間複雜度：O(n)
        空間複雜度：O(h) - h 為樹的高度
        """
        # 基本情況
        if not root:
            return None
        
        # 交換左右子節點
        root.left, root.right = root.right, root.left
        
        # 遞迴翻轉左右子樹
        self.invertTree(root.left)
        self.invertTree(root.right)
        
        return root


# 方法二：更直觀的寫法
class Solution2:
    def invertTree(self, root: Optional[TreeNode]) -> Optional[TreeNode]:
        """
        遞迴法 - 先遞迴再交換
        """
        if not root:
            return None
        
        # 遞迴翻轉左右子樹
        left = self.invertTree(root.left)
        right = self.invertTree(root.right)
        
        # 交換左右子樹
        root.left = right
        root.right = left
        
        return root


# 方法三：迭代法（BFS）
class Solution3:
    def invertTree(self, root: Optional[TreeNode]) -> Optional[TreeNode]:
        """
        迭代法 - 使用佇列
        
        時間複雜度：O(n)
        空間複雜度：O(n)
        """
        if not root:
            return None
        
        queue = deque([root])
        
        while queue:
            node = queue.popleft()
            
            # 交換左右子節點
            node.left, node.right = node.right, node.left
            
            # 將子節點加入佇列
            if node.left:
                queue.append(node.left)
            if node.right:
                queue.append(node.right)
        
        return root


# 方法四：迭代法（使用堆疊）
class Solution4:
    def invertTree(self, root: Optional[TreeNode]) -> Optional[TreeNode]:
        """
        迭代法 - 使用堆疊（DFS）
        """
        if not root:
            return None
        
        stack = [root]
        
        while stack:
            node = stack.pop()
            
            # 交換左右子節點
            node.left, node.right = node.right, node.left
            
            # 將子節點加入堆疊
            if node.left:
                stack.append(node.left)
            if node.right:
                stack.append(node.right)
        
        return root

C++ 解法

#include <queue>
using namespace std;

struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode() : val(0), left(nullptr), right(nullptr) {}
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
    TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
};

class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        // 遞迴法
        if (!root) return nullptr;
        
        // 交換左右子節點
        TreeNode* temp = root->left;
        root->left = root->right;
        root->right = temp;
        
        // 遞迴翻轉左右子樹
        invertTree(root->left);
        invertTree(root->right);
        
        return root;
    }
};

Java 解法

import java.util.*;

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;
    TreeNode() {}
    TreeNode(int val) { this.val = val; }
    TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

class Solution {
    public TreeNode invertTree(TreeNode root) {
        // 遞迴法
        if (root == null) return null;
        
        // 交換左右子節點
        TreeNode temp = root.left;
        root.left = root.right;
        root.right = temp;
        
        // 遞迴翻轉左右子樹
        invertTree(root.left);
        invertTree(root.right);
        
        return root;
    }
}

面試要點

常見追問

  • 能否用迭代法實現？
  • 時間和空間複雜度分別是多少？
  • 如果是 N 叉樹該如何處理？
  • DFS 和 BFS 哪個更好？

解題技巧

  • 遞迴是最簡潔的解法
  • 理解樹的遞迴結構
  • 注意空節點的處理
  • 可以用任何遍歷方式實現

常見錯誤

  • 忘記處理空節點
  • 只交換根節點不遞迴子樹
  • 交換時沒有使用臨時變數
  • 遞迴順序錯誤
