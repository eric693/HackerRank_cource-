設計停車場系統

OOP Design System Design

題目描述

設計一個停車場管理系統，需要考慮不同車型、收費規則、停車位管理等功能。

系統需求：

  1. 支援多種車型（機車、汽車、卡車）
  2. 不同車型需要不同大小的停車位
  3. 實作收費機制（按小時計費）
  4. 追蹤停車位的佔用狀態
  5. 提供車輛進出場的功能

收費標準：

  • 機車：每小時 $10
  • 汽車：每小時 $20
  • 卡車：每小時 $40

功能需求

必須實現的方法：

  • parkVehicle(vehicle)：停放車輛，返回停車位編號
  • removeVehicle(ticketId)：取車，返回停車費用
  • getAvailableSpots()：獲取可用停車位數量
  • getOccupiedSpots()：獲取已佔用停車位數量

設計思路

核心類別設計：

  1. Vehicle（車輛）
    • 屬性：車牌號碼、車型
    • 類型：Motorcycle, Car, Truck
  2. ParkingSpot（停車位）
    • 屬性：編號、大小、是否可用
    • 方法：佔用、釋放
  3. ParkingTicket（停車票）
    • 屬性：票號、車輛、停車位、入場時間
    • 方法：計算費用
  4. ParkingLot（停車場）
    • 管理所有停車位
    • 處理車輛進出
    • 計算費用

Python 實作

from enum import Enum
from datetime import datetime
from typing import Optional

class VehicleType(Enum):
    MOTORCYCLE = 1
    CAR = 2
    TRUCK = 3

class Vehicle:
    def __init__(self, license_plate: str, vehicle_type: VehicleType):
        self.license_plate = license_plate
        self.vehicle_type = vehicle_type
    
    def get_hourly_rate(self) -> int:
        """獲取每小時費率"""
        rates = {
            VehicleType.MOTORCYCLE: 10,
            VehicleType.CAR: 20,
            VehicleType.TRUCK: 40
        }
        return rates[self.vehicle_type]

class ParkingSpot:
    def __init__(self, spot_id: int, spot_type: VehicleType):
        self.spot_id = spot_id
        self.spot_type = spot_type
        self.is_available = True
        self.vehicle: Optional[Vehicle] = None
    
    def can_fit_vehicle(self, vehicle: Vehicle) -> bool:
        """檢查停車位是否適合該車型"""
        # 大停車位可以停小車，小停車位不能停大車
        return self.spot_type.value >= vehicle.vehicle_type.value
    
    def park_vehicle(self, vehicle: Vehicle) -> bool:
        """停放車輛"""
        if not self.is_available:
            return False
        if not self.can_fit_vehicle(vehicle):
            return False
        
        self.vehicle = vehicle
        self.is_available = False
        return True
    
    def remove_vehicle(self):
        """移除車輛"""
        self.vehicle = None
        self.is_available = True

class ParkingTicket:
    _ticket_counter = 0
    
    def __init__(self, vehicle: Vehicle, spot: ParkingSpot):
        ParkingTicket._ticket_counter += 1
        self.ticket_id = ParkingTicket._ticket_counter
        self.vehicle = vehicle
        self.spot = spot
        self.entry_time = datetime.now()
    
    def calculate_fee(self) -> int:
        """計算停車費用"""
        exit_time = datetime.now()
        duration = (exit_time - self.entry_time).total_seconds() / 3600  # 轉換為小時
        hours = max(1, int(duration + 0.5))  # 向上取整，最少 1 小時
        return hours * self.vehicle.get_hourly_rate()

class ParkingLot:
    def __init__(self, motorcycle_spots: int, car_spots: int, truck_spots: int):
        self.spots = []
        self.active_tickets = {}
        
        # 初始化停車位
        spot_id = 1
        for _ in range(motorcycle_spots):
            self.spots.append(ParkingSpot(spot_id, VehicleType.MOTORCYCLE))
            spot_id += 1
        for _ in range(car_spots):
            self.spots.append(ParkingSpot(spot_id, VehicleType.CAR))
            spot_id += 1
        for _ in range(truck_spots):
            self.spots.append(ParkingSpot(spot_id, VehicleType.TRUCK))
            spot_id += 1
    
    def park_vehicle(self, vehicle: Vehicle) -> Optional[int]:
        """停放車輛，返回票號"""
        # 尋找適合的停車位
        for spot in self.spots:
            if spot.is_available and spot.can_fit_vehicle(vehicle):
                if spot.park_vehicle(vehicle):
                    ticket = ParkingTicket(vehicle, spot)
                    self.active_tickets[ticket.ticket_id] = ticket
                    return ticket.ticket_id
        
        return None  # 沒有可用停車位
    
    def remove_vehicle(self, ticket_id: int) -> Optional[int]:
        """取車，返回停車費用"""
        if ticket_id not in self.active_tickets:
            return None
        
        ticket = self.active_tickets[ticket_id]
        fee = ticket.calculate_fee()
        ticket.spot.remove_vehicle()
        del self.active_tickets[ticket_id]
        
        return fee
    
    def get_available_spots(self) -> int:
        """獲取可用停車位數量"""
        return sum(1 for spot in self.spots if spot.is_available)
    
    def get_occupied_spots(self) -> int:
        """獲取已佔用停車位數量"""
        return sum(1 for spot in self.spots if not spot.is_available)


# 使用範例
if __name__ == "__main__":
    # 創建停車場：10個機車位、20個汽車位、5個卡車位
    parking_lot = ParkingLot(10, 20, 5)
    
    # 停放車輛
    car1 = Vehicle("ABC-1234", VehicleType.CAR)
    ticket1 = parking_lot.park_vehicle(car1)
    print(f"車輛已停放，票號：{ticket1}")
    
    motorcycle1 = Vehicle("XYZ-5678", VehicleType.MOTORCYCLE)
    ticket2 = parking_lot.park_vehicle(motorcycle1)
    print(f"機車已停放，票號：{ticket2}")
    
    print(f"可用停車位：{parking_lot.get_available_spots()}")
    print(f"已佔用停車位：{parking_lot.get_occupied_spots()}")
    
    # 取車
    fee = parking_lot.remove_vehicle(ticket1)
    print(f"停車費用：${fee}")
    print(f"可用停車位：{parking_lot.get_available_spots()}")

#include <iostream>
#include <vector>
#include <unordered_map>
#include <ctime>
#include <cmath>

enum class VehicleType {
    MOTORCYCLE = 1,
    CAR = 2,
    TRUCK = 3
};

class Vehicle {
public:
    std::string licensePlate;
    VehicleType vehicleType;
    
    Vehicle(std::string plate, VehicleType type) 
        : licensePlate(plate), vehicleType(type) {}
    
    int getHourlyRate() const {
        switch(vehicleType) {
            case VehicleType::MOTORCYCLE: return 10;
            case VehicleType::CAR: return 20;
            case VehicleType::TRUCK: return 40;
        }
        return 0;
    }
};

class ParkingSpot {
public:
    int spotId;
    VehicleType spotType;
    bool isAvailable;
    Vehicle* vehicle;
    
    ParkingSpot(int id, VehicleType type) 
        : spotId(id), spotType(type), isAvailable(true), vehicle(nullptr) {}
    
    bool canFitVehicle(const Vehicle& v) const {
        return static_cast(spotType) >= static_cast(v.vehicleType);
    }
    
    bool parkVehicle(Vehicle* v) {
        if (!isAvailable || !canFitVehicle(*v)) {
            return false;
        }
        vehicle = v;
        isAvailable = false;
        return true;
    }
    
    void removeVehicle() {
        vehicle = nullptr;
        isAvailable = true;
    }
};

class ParkingTicket {
public:
    int ticketId;
    Vehicle* vehicle;
    ParkingSpot* spot;
    time_t entryTime;
    
    static int ticketCounter;
    
    ParkingTicket(Vehicle* v, ParkingSpot* s) 
        : vehicle(v), spot(s) {
        ticketId = ++ticketCounter;
        entryTime = time(nullptr);
    }
    
    int calculateFee() const {
        time_t exitTime = time(nullptr);
        double hours = difftime(exitTime, entryTime) / 3600.0;
        int billingHours = std::max(1, static_cast(std::ceil(hours)));
        return billingHours * vehicle->getHourlyRate();
    }
};

int ParkingTicket::ticketCounter = 0;

class ParkingLot {
private:
    std::vector<ParkingSpot*> spots;
    std::unordered_map<int, ParkingTicket*> activeTickets;
    
public:
    ParkingLot(int motorcycleSpots, int carSpots, int truckSpots) {
        int spotId = 1;
        for (int i = 0; i < motorcycleSpots; i++) {
            spots.push_back(new ParkingSpot(spotId++, VehicleType::MOTORCYCLE));
        }
        for (int i = 0; i < carSpots; i++) {
            spots.push_back(new ParkingSpot(spotId++, VehicleType::CAR));
        }
        for (int i = 0; i < truckSpots; i++) {
            spots.push_back(new ParkingSpot(spotId++, VehicleType::TRUCK));
        }
    }
    
    int parkVehicle(Vehicle* vehicle) {
        for (auto spot : spots) {
            if (spot->isAvailable && spot->canFitVehicle(*vehicle)) {
                if (spot->parkVehicle(vehicle)) {
                    ParkingTicket* ticket = new ParkingTicket(vehicle, spot);
                    activeTickets[ticket->ticketId] = ticket;
                    return ticket->ticketId;
                }
            }
        }
        return -1;  // 沒有可用停車位
    }
    
    int removeVehicle(int ticketId) {
        if (activeTickets.find(ticketId) == activeTickets.end()) {
            return -1;
        }
        
        ParkingTicket* ticket = activeTickets[ticketId];
        int fee = ticket->calculateFee();
        ticket->spot->removeVehicle();
        activeTickets.erase(ticketId);
        delete ticket;
        
        return fee;
    }
    
    int getAvailableSpots() const {
        int count = 0;
        for (auto spot : spots) {
            if (spot->isAvailable) count++;
        }
        return count;
    }
    
    ~ParkingLot() {
        for (auto spot : spots) {
            delete spot;
        }
        for (auto& pair : activeTickets) {
            delete pair.second;
        }
    }
};

Java 實作

import java.util.*;
import java.time.*;

enum VehicleType {
    MOTORCYCLE(1, 10),
    CAR(2, 20),
    TRUCK(3, 40);
    
    private final int size;
    private final int hourlyRate;
    
    VehicleType(int size, int rate) {
        this.size = size;
        this.hourlyRate = rate;
    }
    
    public int getSize() { return size; }
    public int getHourlyRate() { return hourlyRate; }
}

class Vehicle {
    private String licensePlate;
    private VehicleType vehicleType;
    
    public Vehicle(String plate, VehicleType type) {
        this.licensePlate = plate;
        this.vehicleType = type;
    }
    
    public VehicleType getVehicleType() { return vehicleType; }
    public int getHourlyRate() { return vehicleType.getHourlyRate(); }
}

class ParkingSpot {
    private int spotId;
    private VehicleType spotType;
    private boolean isAvailable;
    private Vehicle vehicle;
    
    public ParkingSpot(int id, VehicleType type) {
        this.spotId = id;
        this.spotType = type;
        this.isAvailable = true;
    }
    
    public boolean canFitVehicle(Vehicle v) {
        return spotType.getSize() >= v.getVehicleType().getSize();
    }
    
    public boolean parkVehicle(Vehicle v) {
        if (!isAvailable || !canFitVehicle(v)) {
            return false;
        }
        this.vehicle = v;
        this.isAvailable = false;
        return true;
    }
    
    public void removeVehicle() {
        this.vehicle = null;
        this.isAvailable = true;
    }
    
    public boolean isAvailable() { return isAvailable; }
}

class ParkingTicket {
    private static int ticketCounter = 0;
    private int ticketId;
    private Vehicle vehicle;
    private ParkingSpot spot;
    private LocalDateTime entryTime;
    
    public ParkingTicket(Vehicle v, ParkingSpot s) {
        this.ticketId = ++ticketCounter;
        this.vehicle = v;
        this.spot = s;
        this.entryTime = LocalDateTime.now();
    }
    
    public int getTicketId() { return ticketId; }
    public ParkingSpot getSpot() { return spot; }
    
    public int calculateFee() {
        LocalDateTime exitTime = LocalDateTime.now();
        long minutes = Duration.between(entryTime, exitTime).toMinutes();
        int hours = Math.max(1, (int) Math.ceil(minutes / 60.0));
        return hours * vehicle.getHourlyRate();
    }
}

class ParkingLot {
    private List<ParkingSpot> spots;
    private Map<Integer, ParkingTicket> activeTickets;
    
    public ParkingLot(int motorcycleSpots, int carSpots, int truckSpots) {
        spots = new ArrayList<>();
        activeTickets = new HashMap<>();
        
        int spotId = 1;
        for (int i = 0; i < motorcycleSpots; i++) {
            spots.add(new ParkingSpot(spotId++, VehicleType.MOTORCYCLE));
        }
        for (int i = 0; i < carSpots; i++) {
            spots.add(new ParkingSpot(spotId++, VehicleType.CAR));
        }
        for (int i = 0; i < truckSpots; i++) {
            spots.add(new ParkingSpot(spotId++, VehicleType.TRUCK));
        }
    }
    
    public Integer parkVehicle(Vehicle vehicle) {
        for (ParkingSpot spot : spots) {
            if (spot.isAvailable() && spot.canFitVehicle(vehicle)) {
                if (spot.parkVehicle(vehicle)) {
                    ParkingTicket ticket = new ParkingTicket(vehicle, spot);
                    activeTickets.put(ticket.getTicketId(), ticket);
                    return ticket.getTicketId();
                }
            }
        }
        return null;
    }
    
    public Integer removeVehicle(int ticketId) {
        ParkingTicket ticket = activeTickets.get(ticketId);
        if (ticket == null) {
            return null;
        }
        
        int fee = ticket.calculateFee();
        ticket.getSpot().removeVehicle();
        activeTickets.remove(ticketId);
        
        return fee;
    }
    
    public int getAvailableSpots() {
        return (int) spots.stream().filter(ParkingSpot::isAvailable).count();
    }
    
    public int getOccupiedSpots() {
        return spots.size() - getAvailableSpots();
    }
}

設計要點

SOLID 原則應用：

  • 單一職責原則：每個類別只負責一個功能
  • 開放封閉原則：容易擴展新的車型
  • 里氏替換原則：不同車型可以互相替換

擴展性考慮：

  • 可以輕鬆添加新的車型
  • 可以修改收費規則（如會員折扣）
  • 可以增加多層停車場
  • 可以添加預約系統
