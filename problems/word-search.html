#79 MEDIUM

Word Search

Array Backtracking Matrix

題目描述

給定一個 m x n 二維字符網格 board 和一個字串 word。如果 word 存在於網格中，返回 true；否則，返回 false。

單詞必須按照字母順序，通過相鄰的單元格內的字母構成，其中「相鄰」單元格是那些水平相鄰或垂直相鄰的單元格。同一個單元格內的字母不允許被重複使用。

範例

範例 1:

輸入：board = [["A","B","C","E"],["S","F","C","S"],["A","D","E","E"]], word = "ABCCED"
輸出：true

視覺化：
A→B→C←C
S F↓C S
A D E→E→D

範例 2:

輸入：board = [["A","B","C","E"],["S","F","C","S"],["A","D","E","E"]], word = "SEE"
輸出：true

視覺化：
A B C E
S F C→S
A D→E→E

範例 3:

輸入：board = [["A","B","C","E"],["S","F","C","S"],["A","D","E","E"]], word = "ABCB"
輸出：false
解釋：字母 'B' 不能被重複使用。

限制條件

  • m == board.length
  • n == board[i].length
  • 1 <= m, n <= 6
  • 1 <= word.length <= 15
  • board 和 word 僅由大小寫英文字母組成

解題思路

回溯法（DFS）

這是回溯算法的經典應用：

  1. 遍歷網格，找到與單詞首字母匹配的位置
  2. 從該位置開始 DFS，向四個方向搜索
  3. 標記當前位置為已訪問（避免重複使用）
  4. 如果找到完整單詞，返回 true
  5. 回溯：恢復當前位置的狀態

優化技巧

  • 提前終止：如果當前字符不匹配，立即返回
  • 原地標記：可以直接修改 board 來標記訪問狀態
  • 剪枝：統計字符頻率，如果 board 中某字符數量少於 word，直接返回 false
  • 時間複雜度：O(m × n × 4^L) - L 為單詞長度
  • 空間複雜度：O(L) - 遞迴深度

程式碼實作

Python 解法

class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        """
        回溯法（DFS）
        
        時間複雜度：O(m × n × 4^L)
        空間複雜度：O(L)
        """
        if not board or not board[0]:
            return False
        
        rows, cols = len(board), len(board[0])
        
        def backtrack(r, c, index):
            # 找到完整單詞
            if index == len(word):
                return True
            
            # 邊界檢查和字符匹配檢查
            if (r < 0 or r >= rows or c < 0 or c >= cols or 
                board[r][c] != word[index]):
                return False
            
            # 標記當前位置為已訪問
            temp = board[r][c]
            board[r][c] = '#'
            
            # 向四個方向搜索
            found = (backtrack(r + 1, c, index + 1) or
                    backtrack(r - 1, c, index + 1) or
                    backtrack(r, c + 1, index + 1) or
                    backtrack(r, c - 1, index + 1))
            
            # 回溯：恢復狀態
            board[r][c] = temp
            
            return found
        
        # 從每個位置開始嘗試
        for r in range(rows):
            for c in range(cols):
                if board[r][c] == word[0]:
                    if backtrack(r, c, 0):
                        return True
        
        return False


# 方法二：使用 visited 集合
class Solution2:
    def exist(self, board: List[List[str]], word: str) -> bool:
        """
        使用 visited 集合（不修改原陣列）
        """
        rows, cols = len(board), len(board[0])
        
        def backtrack(r, c, index, visited):
            if index == len(word):
                return True
            
            if (r < 0 or r >= rows or c < 0 or c >= cols or 
                (r, c) in visited or board[r][c] != word[index]):
                return False
            
            visited.add((r, c))
            
            found = (backtrack(r + 1, c, index + 1, visited) or
                    backtrack(r - 1, c, index + 1, visited) or
                    backtrack(r, c + 1, index + 1, visited) or
                    backtrack(r, c - 1, index + 1, visited))
            
            visited.remove((r, c))
            
            return found
        
        for r in range(rows):
            for c in range(cols):
                if board[r][c] == word[0]:
                    if backtrack(r, c, 0, set()):
                        return True
        
        return False

C++ 解法

#include <vector>
#include <string>
using namespace std;

class Solution {
public:
    bool exist(vector<vector<char>>& board, string word) {
        if (board.empty() || board[0].empty()) return false;
        
        rows = board.size();
        cols = board[0].size();
        
        for (int r = 0; r < rows; r++) {
            for (int c = 0; c < cols; c++) {
                if (board[r][c] == word[0]) {
                    if (backtrack(board, word, r, c, 0)) {
                        return true;
                    }
                }
            }
        }
        
        return false;
    }
    
private:
    int rows, cols;
    
    bool backtrack(vector<vector<char>>& board, const string& word, 
                   int r, int c, int index) {
        if (index == word.length()) return true;
        
        if (r < 0 || r >= rows || c < 0 || c >= cols || 
            board[r][c] != word[index]) {
            return false;
        }
        
        char temp = board[r][c];
        board[r][c] = '#';
        
        bool found = backtrack(board, word, r + 1, c, index + 1) ||
                    backtrack(board, word, r - 1, c, index + 1) ||
                    backtrack(board, word, r, c + 1, index + 1) ||
                    backtrack(board, word, r, c - 1, index + 1);
        
        board[r][c] = temp;
        
        return found;
    }
};

Java 解法

class Solution {
    private int rows, cols;
    
    public boolean exist(char[][] board, String word) {
        if (board == null || board.length == 0) return false;
        
        rows = board.length;
        cols = board[0].length;
        
        for (int r = 0; r < rows; r++) {
            for (int c = 0; c < cols; c++) {
                if (board[r][c] == word.charAt(0)) {
                    if (backtrack(board, word, r, c, 0)) {
                        return true;
                    }
                }
            }
        }
        
        return false;
    }
    
    private boolean backtrack(char[][] board, String word, 
                             int r, int c, int index) {
        if (index == word.length()) return true;
        
        if (r < 0 || r >= rows || c < 0 || c >= cols || 
            board[r][c] != word.charAt(index)) {
            return false;
        }
        
        char temp = board[r][c];
        board[r][c] = '#';
        
        boolean found = backtrack(board, word, r + 1, c, index + 1) ||
                       backtrack(board, word, r - 1, c, index + 1) ||
                       backtrack(board, word, r, c + 1, index + 1) ||
                       backtrack(board, word, r, c - 1, index + 1);
        
        board[r][c] = temp;
        
        return found;
    }
}

面試要點

常見追問

  • 如何找出所有可能的路徑？
  • 能否不修改原陣列？
  • 如何優化性能？
  • 二維矩陣和圖的 DFS 有何不同？

解題技巧

  • 回溯三步：選擇、遞迴、撤銷
  • 原地標記避免額外空間
  • 四個方向的遍歷
  • 提前終止優化性能

常見錯誤

  • 忘記回溯（恢復狀態）
  • 邊界檢查遺漏
  • 沒有標記已訪問導致重複使用
  • 終止條件寫錯
