#297 HARD

Serialize and Deserialize Binary Tree

Tree DFS BFS Design String

題目描述

序列化是將一個資料結構或者物件轉換為連續的比特位的操作，進而可以將轉換後的資料存儲在一個檔案或者記憶體中，同時也可以通過網路傳輸到另一個電腦環境，採取相反方式重構得到原資料。

請設計一個算法來實現二元樹的序列化與反序列化。這裡不限定你的序列/反序列化算法執行邏輯，你只需要保證一個二元樹可以被序列化為一個字串並且將這個字串反序列化為原始的樹結構。

範例

範例 1:

輸入：root = [1,2,3,null,null,4,5]
輸出：[1,2,3,null,null,4,5]

範例 2:

輸入：root = []
輸出：[]

限制條件

  • 樹中節點數在範圍 [0, 104] 內
  • -1000 <= Node.val <= 1000

解題思路

方法一：前序遍歷（DFS）- 推薦

序列化：使用前序遍歷將樹轉換為字串

  • 遍歷順序：根節點 → 左子樹 → 右子樹
  • 使用特殊標記（如 "null"）表示空節點
  • 節點之間用分隔符（如逗號）分隔

反序列化：根據字串重建樹

  • 將字串分割成節點列表
  • 使用遞迴按前序遍歷順序重建
  • 遇到 "null" 則返回空節點
  • 時間複雜度：O(n)
  • 空間複雜度：O(n)

方法二：層序遍歷（BFS）

使用隊列進行層序遍歷，記錄每層的節點。

# Definition for a binary tree node.
class TreeNode(object):
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None

class Codec:
    """
    前序遍歷 DFS 解法
    
    時間複雜度：O(n)
    空間複雜度：O(n)
    """
    
    def serialize(self, root):
        """將樹序列化為字符串
        
        Args:
            root: TreeNode
        Returns:
            str
        """
        def dfs(node):
            if not node:
                return "null,"
            # 前序遍歷：根 -> 左 -> 右
            return str(node.val) + "," + dfs(node.left) + dfs(node.right)
        
        return dfs(root)
    
    def deserialize(self, data):
        """將字符串反序列化為樹
        
        Args:
            data: str
        Returns:
            TreeNode
        """
        def dfs():
            val = next(vals)
            if val == "null":
                return None
            # 重建節點
            node = TreeNode(int(val))
            node.left = dfs()
            node.right = dfs()
            return node
        
        vals = iter(data.split(","))
        return dfs()


# 方法二：BFS 層序遍歷
class Codec2:
    def serialize(self, root):
        if not root:
            return ""
        
        from collections import deque
        queue = deque([root])
        result = []
        
        while queue:
            node = queue.popleft()
            if node:
                result.append(str(node.val))
                queue.append(node.left)
                queue.append(node.right)
            else:
                result.append("null")
        
        return ",".join(result)
    
    def deserialize(self, data):
        if not data:
            return None
        
        from collections import deque
        vals = data.split(",")
        root = TreeNode(int(vals[0]))
        queue = deque([root])
        i = 1
        
        while queue and i < len(vals):
            node = queue.popleft()
            
            if vals[i] != "null":
                node.left = TreeNode(int(vals[i]))
                queue.append(node.left)
            i += 1
            
            if i < len(vals) and vals[i] != "null":
                node.right = TreeNode(int(vals[i]))
                queue.append(node.right)
            i += 1
        
        return root

C++ 解法

#include <string>
#include <queue>
#include <sstream>
using namespace std;

struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

class Codec {
public:
    // 序列化
    string serialize(TreeNode* root) {
        if (!root) return "null,";
        
        return to_string(root->val) + "," +
               serialize(root->left) +
               serialize(root->right);
    }

    // 反序列化
    TreeNode* deserialize(string data) {
        queue<string> nodes;
        string val;
        
        // 分割字符串
        for (char c : data) {
            if (c == ',') {
                nodes.push(val);
                val.clear();
            } else {
                val += c;
            }
        }
        
        return helper(nodes);
    }
    
private:
    TreeNode* helper(queue<string>& nodes) {
        string val = nodes.front();
        nodes.pop();
        
        if (val == "null") return nullptr;
        
        TreeNode* node = new TreeNode(stoi(val));
        node->left = helper(nodes);
        node->right = helper(nodes);
        return node;
    }
};

Java 解法

import java.util.*;

public class Codec {
    
    // 序列化
    public String serialize(TreeNode root) {
        if (root == null) {
            return "null,";
        }
        
        return root.val + "," +
               serialize(root.left) +
               serialize(root.right);
    }

    // 反序列化
    public TreeNode deserialize(String data) {
        Queue<String> nodes = new LinkedList<>(
            Arrays.asList(data.split(","))
        );
        return helper(nodes);
    }
    
    private TreeNode helper(Queue<String> nodes) {
        String val = nodes.poll();
        
        if (val.equals("null")) {
            return null;
        }
        
        TreeNode node = new TreeNode(Integer.parseInt(val));
        node.left = helper(nodes);
        node.right = helper(nodes);
        return node;
    }
}

算法過程視覺化

序列化過程（前序遍歷）：

原始樹：
      1
     / \
    2   3
       / \
      4   5

前序遍歷順序：1 -> 2 -> 3 -> 4 -> 5

序列化結果：
"1,2,null,null,3,4,null,null,5,null,null,"

解析：
1,              # 根節點 1
2,null,null,    # 左子樹：節點 2（無左右子節點）
3,              # 右子樹：節點 3
4,null,null,    # 3 的左子節點 4
5,null,null     # 3 的右子節點 5

反序列化過程：

輸入："1,2,null,null,3,4,null,null,5,null,null,"

步驟：
1. 讀取 "1" → 創建根節點 1
2. 遞迴處理左子樹：
   - 讀取 "2" → 創建節點 2
   - 讀取 "null" → 左子節點為空
   - 讀取 "null" → 右子節點為空
3. 遞迴處理右子樹：
   - 讀取 "3" → 創建節點 3
   - 遞迴左子樹：讀取 "4" → 創建節點 4
   - 遞迴右子樹：讀取 "5" → 創建節點 5

重建完成！

面試要點

常見追問

  • 為什麼選擇前序遍歷？
  • 能否用其他遍歷方式？
  • 如何處理特殊字符？
  • 空間優化的可能性？

解題技巧

  • 前序遍歷便於重建根節點
  • 使用分隔符避免歧義
  • null 標記表示空節點
  • 遞迴實現簡潔清晰

常見錯誤

  • 忘記處理空節點
  • 分隔符選擇不當
  • 遞迴終止條件錯誤
  • 字符串分割問題

為什麼使用前序遍歷？

前序遍歷的優勢：

  • 根節點在最前面，便於首先創建根節點
  • 然後依次處理左右子樹，符合遞迴邏輯
  • 重建時可以立即確定樹的結構

其他遍歷方式：

  • 後序遍歷：也可以，但需要從後往前讀取
  • 中序遍歷：單獨使用無法唯一確定樹結構
  • 層序遍歷：可以，但實現稍複雜
